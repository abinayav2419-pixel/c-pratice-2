#include <stdio.h>

int main() {
    int N;
    scanf("%d", &N);

    int transactions[N];
    for(int i = 0; i < N; i++) {
        scanf("%d", &transactions[i]);
    }

    int i = 0;
    int consecutiveHigh = 0;
    int highValueCount = 0;
    int fraudAttempt = -1;

    while(i < N) {
        if(transactions[i] >= 50000) {
            consecutiveHigh++;
            highValueCount++;
            if(consecutiveHigh == 3 && fraudAttempt == -1) {
                fraudAttempt = i + 1;
            }
        } else {
            consecutiveHigh = 0;
        }
        i++;
    }

    if(fraudAttempt != -1)
        printf("Fraud Triggered At Attempt: %d\n", fraudAttempt);
    else
        printf("Fraud Triggered At Attempt: Not Triggered\n");

    printf("High-Value Transactions: %d\n", highValueCount);

    return 0;
}
