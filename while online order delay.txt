#include <stdio.h>

int main() {
    int N;
    scanf("%d", &N);

    int delays[N];
    int i = 0;
    int total_delay = 0;
    int delayed_days = 0;

    
    while (i < N) {
        scanf("%d", &delays[i]);
        i++;
    }

    i = 0;  

   
    while (i < N) {
        total_delay += delays[i];
        if (delays[i] > 2) {
            delayed_days++;
        }
        i++;
    }

  
    printf("Total Delay: %d\n", total_delay);
    printf("Delayed Days: %d\n", delayed_days);

    return 0;
}
