#include <stdio.h>

int main() {
    int fuel, N;
    
 
    scanf("%d", &fuel);
    scanf("%d", &N);
    
    int consumption[N];
    
   
    for(int i = 0; i < N; i++) {
        scanf("%d", &consumption[i]);
    }
    
    int completedTrips = 0;
    int i = 0;
    
   
    while(i < N && fuel >= consumption[i]) {
        fuel -= consumption[i];
        completedTrips++;
        i++;
    }
    
   
    printf("Completed Trips: %d\n", completedTrips);
    printf("Remaining Fuel: %d\n", fuel);
    
    return 0;
}
