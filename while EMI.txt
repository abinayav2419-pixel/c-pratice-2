#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);

    int arr[n];
    int i = 0;
    while(i < n) {
        scanf("%d", &arr[i]);
        i++;
    }

    int maxStreak = 0;
    int currentStreak = 0;
    i = 0;


    while(i < n) {
        if(arr[i] == 0) {
            currentStreak++;
            if(currentStreak > maxStreak) {
                maxStreak = currentStreak;
            }
        } else {
            currentStreak = 0;
        }
        i++;
    }

    printf("Longest Defult Streak: %d", maxStreak);

    return 0;
}
