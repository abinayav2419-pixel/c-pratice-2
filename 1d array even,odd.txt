#include <stdio.h>

int main() {
    int n, i;
    
    printf("Enter how many numbers: ");
    scanf("%d", &n);

    int arr[n];   // array declaration

    printf("Enter %d numbers:\n", n);
    for(i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    int even = 0, odd = 0;

    for(i = 0; i < n; i++) {
        if(arr[i] % 2 == 0)
            even++;
        else
            odd++;
    }

    printf("Even: %d ", even);
    printf("Odd: %d", odd);

    return 0;
}
