#include <stdio.h>

int main() {
    int N;
    scanf("%d", &N);

    int overtime[N];
    int i = 0;
    int total_overtime = 0;
    int burnout_days = 0;

    
    while (i < N) {
        scanf("%d", &overtime[i]);
        i++;
    }

    i = 0;  

   
    while (i < N) {
        total_overtime += overtime[i];
        if (overtime[i] > 4) {
            burnout_days++;
        }
        i++;
    }

  
    printf("Total Overtime: %d\n", total_overtime);
    printf("Burnout Days: %d\n", burnout_days);

    return 0;
}
