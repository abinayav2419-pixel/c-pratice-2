#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n);

    int successful = 0, cancelled = 0;
    int i = 0;
    int order;

    while (i < n) {
        scanf("%d", &order);

        if (order == 1) {
            successful++;
        } 
        else if (order == 0) {
            cancelled++;
        }

        i++;
    }

    printf("Successful: %d\n", successful);
    printf("Cancelled: %d\n", cancelled);

    if (cancelled > successful) {
        printf("Status: Risk\n");
    } else {
        printf("Status: Safe\n");
    }

    return 0;
}
