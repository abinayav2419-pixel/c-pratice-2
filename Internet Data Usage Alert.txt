# include <stdio.h>
int main(){
    int i,n;
    int data,total = 0; 
     int highdays = 0;
    
    scanf("%d",&n);
    for(i=0;i<n;i++){
      scanf("%d",&data);
      total+= data;
      
      if(data>2){
          highdays++;
      }
}

    printf("Total Usage:%d\n",total);
    printf("High Usage Days:%d",highdays);
    return 0;
}
