#include <stdio.h>
int main() {
    int n, X;
    scanf("%d", &n);

    int arr[n];
    for(int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }
    scanf("%d", &X);

    int count = 0;
    for(int i = 0; i < n; i++) {
        if(arr[i] == X) {
            count++;
        }
    }
    printf("%d\n", count);

    return 0;
}
