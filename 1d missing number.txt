#include <stdio.h>

int main() {
    int n;

    printf("Enter value of N: ");
    scanf("%d", &n);

    int arr[n-1];
    int sum = 0;

    printf("Enter %d numbers:\n", n-1);
    for(int i = 0; i < n-1; i++) {
        scanf("%d", &arr[i]);
        sum += arr[i];
    }

    int expected_sum = n * (n + 1) / 2;
    int missing = expected_sum - sum;

    printf("Missing number is: %d\n", missing);

    return 0;
}
