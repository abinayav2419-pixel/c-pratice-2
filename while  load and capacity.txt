#include <stdio.h>

int main() {
    int capacity, N;
    
    scanf("%d", &capacity);
    scanf("%d", &N);
    
    int load[N];
    for(int i = 0; i < N; i++) {
        scanf("%d", &load[i]);
    }
    
    int safeHours = 0;
    int failureCount = 0;
    int i = 0;

    while(i < N) {
        if(load[i] > capacity) {
            failureCount++;
        } else {
            safeHours++;
        }
        i++;
    }
    
    printf("Safe Hours: %d\n", safeHours);
    printf("Failure Count: %d\n", failureCount);
    
    return 0;
}
