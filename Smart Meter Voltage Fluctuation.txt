# include <stdio.h>
int main(){
    int i,n;
    int voltage,min = 0; 
     int low = 0;
    
    scanf("%d",&n);
    
    for(i=0;i<n;i++){
      scanf("%d",&voltage);
     if(i==0){
         min = voltage;
     }
      
      if(voltage<min){
          min=voltage;
      }
      if(voltage<210){
          low++;
      }
}

    printf("Minimum Voltage :%d\n",min);
    printf("Low Voltage Events:%d",low);
    return 0;
}
