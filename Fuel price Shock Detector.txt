# include <stdio.h>
int main(){
    int i,n;
    int price,max = 0; 
     int highdays = 0;
    
    scanf("%d",&n);
    
    for(i=0;i<n;i++){
      scanf("%d",& price);
     if(price>max){
         max= price;
     }
      
      if(price>=100){
          highdays++;
      }
}

    printf("Highest price :%d\n",max);
    printf("High Price Days:%d",highdays);
    return 0;
}
