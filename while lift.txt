#include <stdio.h>

int main() {
    int maxWeight, N;
    
    
    scanf("%d", &maxWeight);
    scanf("%d", &N);
    
    int weights[N];
    for(int i = 0; i < N; i++) {
        scanf("%d", &weights[i]);
    }
    
    int totalWeight = 0;
    int count = 0;
    int i = 0;
    int overload = 0;  // 0 = No, 1 = Yes
    
    
    while(i < N) {
        if(totalWeight + weights[i] > maxWeight) {
            overload = 1;
            break;
        }
        
        totalWeight += weights[i];
        count++;
        i++;
    }
    
   
    printf("Passengers Allowed: %d\n", count);
    
    if(overload)
        printf("Overload: Yes\n");
    else
        printf("Overload: No\n");

    return 0;
}
