# include <stdio.h>
int main() {
    int cash, N;
scanf("%d", &cash);
    scanf("%d", &N);
    
    int consumption[N];
    
   
    for(int i = 0; i < N; i++) {
        scanf("%d", &consumption[i]);
    }
    
    int final_balance = 0;
    int i = 0;
    
   
    while(i < N && cash >= consumption[i]) {
        cash-= consumption[i];
        final_balance++;
        i++;
    }
    
   
    printf("Successful Purchases: %d\n", final_balance);
    printf("Final Balance: %d\n", cash);
    
    return 0;
}
