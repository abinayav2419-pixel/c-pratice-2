#include <stdio.h>

void rotateLeft(int arr[], int n, int k) {
    int temp[n];

    k = k % n;   

    for(int i = 0; i < n; i++) {
        temp[i] = arr[(i + k) % n];
    }

    for(int i = 0; i < n; i++) {
        arr[i] = temp[i];
    }
}

int main() {
    int n, k;

    printf("Enter number of elements: ");
    scanf("%d", &n);

    int arr[n];

    printf("Enter %d elements:\n", n);
    for(int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    printf("Enter value of K: ");
    scanf("%d", &k);

    rotateLeft(arr, n, k);

    printf("Array after left rotation:\n");
    for(int i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }

    return 0;
}
